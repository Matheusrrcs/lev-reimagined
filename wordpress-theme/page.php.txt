<?php
/**
 * Generic Page Template
 *
 * @package Clinica_Levin
 */

get_header();
?>

<section class="section-lg" style="padding-top: 120px;">
    <div class="container">
        <?php while (have_posts()) : the_post(); ?>
        
        <article id="post-<?php the_ID(); ?>" <?php post_class(); ?>>
            <header class="section-header animate-fadeInUp">
                <h1 class="section-title"><?php the_title(); ?></h1>
                <?php if (has_excerpt()) : ?>
                <p class="section-description"><?php the_excerpt(); ?></p>
                <?php endif; ?>
            </header>

            <?php if (has_post_thumbnail()) : ?>
            <div class="mb-5 animate-fadeInUp delay-100">
                <img src="<?php the_post_thumbnail_url('large'); ?>" 
                     alt="<?php the_title_attribute(); ?>" 
                     class="hero-image" 
                     style="width: 100%; max-height: 500px; object-fit: cover;">
            </div>
            <?php endif; ?>

            <div class="page-content animate-fadeInUp delay-200">
                <?php the_content(); ?>
            </div>

            <?php
            // If comments are open or we have at least one comment, load up the comment template
            if (comments_open() || get_comments_number()) :
                comments_template();
            endif;
            ?>
        </article>

        <?php endwhile; ?>
    </div>
</section>

<?php get_footer(); ?>
