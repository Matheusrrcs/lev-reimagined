<?php
/**
 * Single Post Template
 *
 * @package Clinica_Levin
 */

get_header();
?>

<article class="section-lg" style="padding-top: 120px;">
    <div class="container" style="max-width: 800px;">
        <?php while (have_posts()) : the_post(); ?>
        
        <!-- Post Header -->
        <header class="animate-fadeInUp" style="margin-bottom: var(--spacing-xl);">
            <div style="display: flex; gap: var(--spacing-sm); margin-bottom: var(--spacing-md);">
                <?php
                $categories = get_the_category();
                foreach ($categories as $category) :
                ?>
                <span class="section-badge"><?php echo esc_html($category->name); ?></span>
                <?php endforeach; ?>
            </div>
            
            <h1 class="section-title" style="margin-bottom: var(--spacing-md);"><?php the_title(); ?></h1>
            
            <div style="display: flex; align-items: center; gap: var(--spacing-md); color: hsl(var(--muted-foreground)); font-size: 0.875rem;">
                <span style="display: flex; align-items: center; gap: 0.5rem;">
                    <svg xmlns="http://www.w3.org/2000/svg" width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><rect width="18" height="18" x="3" y="4" rx="2" ry="2"/><line x1="16" x2="16" y1="2" y2="6"/><line x1="8" x2="8" y1="2" y2="6"/><line x1="3" x2="21" y1="10" y2="10"/></svg>
                    <?php echo get_the_date(); ?>
                </span>
                <span style="display: flex; align-items: center; gap: 0.5rem;">
                    <svg xmlns="http://www.w3.org/2000/svg" width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><circle cx="12" cy="12" r="10"/><polyline points="12 6 12 12 16 14"/></svg>
                    <?php echo ceil(str_word_count(get_the_content()) / 200); ?> min de leitura
                </span>
            </div>
        </header>

        <!-- Featured Image -->
        <?php if (has_post_thumbnail()) : ?>
        <div class="animate-fadeInUp delay-100" style="margin-bottom: var(--spacing-xl);">
            <img src="<?php the_post_thumbnail_url('large'); ?>" 
                 alt="<?php the_title_attribute(); ?>" 
                 style="width: 100%; border-radius: var(--radius-xl); box-shadow: var(--shadow-lg);">
        </div>
        <?php endif; ?>

        <!-- Post Content -->
        <div class="post-content animate-fadeInUp delay-200" style="font-size: 1.125rem; line-height: 1.8;">
            <?php the_content(); ?>
        </div>

        <!-- Tags -->
        <?php
        $tags = get_the_tags();
        if ($tags) :
        ?>
        <div style="margin-top: var(--spacing-xl); padding-top: var(--spacing-lg); border-top: 1px solid hsl(var(--border));">
            <div style="display: flex; flex-wrap: wrap; gap: 0.5rem;">
                <?php foreach ($tags as $tag) : ?>
                <a href="<?php echo get_tag_link($tag->term_id); ?>" 
                   style="font-size: 0.875rem; padding: 0.5rem 1rem; background: hsl(var(--muted)); border-radius: var(--radius-full); color: hsl(var(--muted-foreground)); transition: all 0.3s ease;">
                    #<?php echo esc_html($tag->name); ?>
                </a>
                <?php endforeach; ?>
            </div>
        </div>
        <?php endif; ?>

        <!-- Author Box -->
        <div class="card" style="margin-top: var(--spacing-xl); display: flex; gap: var(--spacing-lg); align-items: center;">
            <?php echo get_avatar(get_the_author_meta('ID'), 80, '', '', array('class' => 'avatar', 'extra_attr' => 'style="border-radius: 50%;"')); ?>
            <div>
                <h4 style="margin-bottom: 0.25rem;">Escrito por <?php the_author(); ?></h4>
                <p class="card-text" style="margin-bottom: 0;"><?php echo get_the_author_meta('description'); ?></p>
            </div>
        </div>

        <!-- Post Navigation -->
        <nav style="margin-top: var(--spacing-xl); display: grid; grid-template-columns: 1fr 1fr; gap: var(--spacing-lg);">
            <?php
            $prev_post = get_previous_post();
            $next_post = get_next_post();
            
            if ($prev_post) :
            ?>
            <a href="<?php echo get_permalink($prev_post); ?>" class="card" style="text-decoration: none;">
                <span style="font-size: 0.75rem; color: hsl(var(--muted-foreground)); text-transform: uppercase; letter-spacing: 0.05em;">
                    ← Anterior
                </span>
                <h4 class="card-title" style="margin-top: 0.5rem; font-size: 1rem;">
                    <?php echo get_the_title($prev_post); ?>
                </h4>
            </a>
            <?php else : ?>
            <div></div>
            <?php endif; ?>
            
            <?php if ($next_post) : ?>
            <a href="<?php echo get_permalink($next_post); ?>" class="card" style="text-decoration: none; text-align: right;">
                <span style="font-size: 0.75rem; color: hsl(var(--muted-foreground)); text-transform: uppercase; letter-spacing: 0.05em;">
                    Próximo →
                </span>
                <h4 class="card-title" style="margin-top: 0.5rem; font-size: 1rem;">
                    <?php echo get_the_title($next_post); ?>
                </h4>
            </a>
            <?php endif; ?>
        </nav>

        <?php endwhile; ?>
    </div>
</article>

<!-- Related Posts -->
<?php
$categories = get_the_category();
if ($categories) :
    $category_ids = array();
    foreach ($categories as $category) {
        $category_ids[] = $category->term_id;
    }
    
    $related_posts = new WP_Query(array(
        'category__in'   => $category_ids,
        'post__not_in'   => array(get_the_ID()),
        'posts_per_page' => 3,
        'orderby'        => 'rand',
    ));
    
    if ($related_posts->have_posts()) :
?>
<section class="about-section section-lg">
    <div class="container">
        <div class="section-header animate-on-scroll">
            <h2 class="section-title">Artigos Relacionados</h2>
        </div>
        
        <div class="services-grid">
            <?php while ($related_posts->have_posts()) : $related_posts->the_post(); ?>
            <article class="card animate-on-scroll">
                <?php if (has_post_thumbnail()) : ?>
                <img src="<?php the_post_thumbnail_url('blog-card'); ?>" alt="<?php the_title_attribute(); ?>" class="card-image">
                <?php endif; ?>
                
                <h3 class="card-title"><?php the_title(); ?></h3>
                <p class="card-text"><?php echo wp_trim_words(get_the_excerpt(), 15); ?></p>
                
                <a href="<?php the_permalink(); ?>" class="btn btn-outline mt-3">
                    Ler Artigo
                    <svg xmlns="http://www.w3.org/2000/svg" width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M5 12h14"/><path d="m12 5 7 7-7 7"/></svg>
                </a>
            </article>
            <?php endwhile; wp_reset_postdata(); ?>
        </div>
    </div>
</section>
<?php
    endif;
endif;
?>

<!-- CTA -->
<section class="cta-section section-lg">
    <div class="container">
        <div class="cta-content animate-on-scroll">
            <h2 class="cta-title">Gostou do Conteúdo?</h2>
            <p class="cta-description">
                Agende uma consulta e receba orientação personalizada de nossos especialistas.
            </p>
            <div class="cta-buttons">
                <?php
                $whatsapp = get_theme_mod('whatsapp_number', '');
                if ($whatsapp) :
                ?>
                <a href="https://wa.me/<?php echo esc_attr($whatsapp); ?>?text=<?php echo urlencode('Olá! Li um artigo no blog e gostaria de agendar uma consulta.'); ?>" 
                   class="btn btn-white btn-lg" 
                   target="_blank" 
                   rel="noopener noreferrer">
                    Agendar Consulta
                </a>
                <?php endif; ?>
            </div>
        </div>
    </div>
</section>

<?php get_footer(); ?>
